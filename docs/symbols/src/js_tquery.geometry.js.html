1 /**
  2  * Handle geometry
  3  *
  4  * @class include THREE.Geometry
  5  *
  6  * @param {THREE.Geometry} object an instance or an array of instance
  7 */
  8 tQuery.Geometry	= function(object){
  9 	this._lists	= object instanceof Array ? object : [object];
 10 	this.length	= this._lists.length;
 11 	// sanity check - all items MUST be THREE.Geometery
 12 	this._lists.forEach(function(item){ console.assert(item instanceof THREE.Geometry);	});
 13 	
 14 	// compute bounding boxes
 15 	this.each(function(geometry){
 16 		geometry.computeBoundingBox();
 17 	});
 18 };
 19 
 20 // make it pluginable
 21 tQuery.pluginsMixin(tQuery.Geometry);

 32 /**
 33  * Retrieve the elements matched by the jQuery object
 34  * 
 35  * @param {Function} callback the function to notify. function(element){ }.
 36  * 			loop interrupted if it returns false
 37  * 
 38  * @returns {Boolean} return true if completed, false if interrupted
 39 */
 40 tQuery.Geometry.prototype.get	= function(idx){
 41 	if( idx === undefined )	return this._lists;
 42 
 43 	// sanity check - it MUST be defined
 44 	console.assert(this._lists[idx], "element not defined");
 45 	return this._lists[idx];
 46 };
 47 
 48 /**
 49  * loop over element
 50  * 
 51  * @param {Function} callback the function to notify. function(element){ }.
 52  * 			loop interrupted if it returns false
 53  * 
 54  * @returns {Boolean} return true if completed, false if interrupted
 55 */
 56 tQuery.Geometry.prototype.each	= function(callback){
 57 	for(var i = 0; i < this._lists.length; i++){
 58 		var object3d	= this._lists[i];
 59 		var keepLooping	= callback(object3d)
 60 		if( keepLooping === false )	return false;
 61 	}
 62 	return true;
 63 };

 65 /**
 66  * getter/setter of the back pointer
 67  *
 68  * @param {Object} back the value to return when .back() is called. optional
 69  * 
 70 */
 71 tQuery.Geometry.prototype.back	= function(value){
 72 	if( value === undefined )	return this._back;
 73 	this._back	= value;
 74 	return this;
 75 };
