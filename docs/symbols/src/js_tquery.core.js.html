  5 /**
  6  * Create a tQuery element
  7  *
  8  * @class root class
  9  * 
 10  * @param {} object
 11  * @param {THREE.Object3D} rootnode
 12  * @returns {tQuery.*} the tQuery object created
 13 */
 14 var tQuery	= function(object, root)
 15 {
 16 	if( object instanceof THREE.Object3D  && tQuery.Object3D){
 17 		return new tQuery.Object3D(object);
 18 	}else if( object instanceof THREE.Geometry && tQuery.Geometry){
 19 		return new tQuery.Geometry(object);
 20 	}else if( object instanceof THREE.Material && tQuery.Material){
 21 		return new tQuery.Material(object);
 22 	}else if( typeof object === "string" && tQuery.Object3D){
 23 		return new tQuery.Object3D(object, root);
 24 	}else{
 25 		console.assert(false, "unsupported type")
 26 	}
 27 	return undefined;
 28 };

 39 /**
 40  * generic getter/setter
 41  * 
 42  * @param {Object} object the object in which store the data
 43  * @param {String} key the key/name of the data to get/set
 44  * @param {*} value the value to set (optional)
 45  * 
 46  * @returns {*} return the value stored in this object for this key
 47 */
 48 tQuery.data	= function(object, key, value)
 49 {
 50 	// sanity check
 51 	console.assert( object, 'invalid parameters' );
 52 	console.assert( typeof key === 'string', 'invalid parameters');
 53 
 54 	// init _tqData
 55 	object['_tqData']	= object['_tqData']	|| {};
 56 	// set the value if any
 57 	if( value ){
 58 		object['_tqData'][key]	= value;
 59 	}
 60 	// return the value
 61 	return object['_tqData'][key];
 62 };
 63 
 64 /**
 65  * Same as jQuery.removeData()
 66 */
 67 tQuery.removeData	= function(object, key)
 68 {
 69 	// handle the 'key as Array' case
 70 	if( key instanceof Array ){
 71 		key.forEach(function(key){
 72 			tQuery.removeData(object, key);
 73 		})
 74 		return;
 75 	}
 76 	// sanity check
 77 	console.assert( typeof key === "string");
 78 	// do delete the key
 79 	delete object['_tqData'][key];
 80 }

 87 /**
 88  * loop over a Array.
 89  * 
 90  * @param {Array} arr the array to traverse.
 91  * @param {Function} callback the function to notify. function(element){ }.
 92  * 			loop interrupted if it returns false
 93  * 
 94  * @returns {Boolean} return true if completed, false if interrupted
 95 */
 96 tQuery.each	= function(arr, callback){
 97 	for(var i = 0; i < arr.length; i++){
 98 		var keepLooping	= callback(arr[i])
 99 		if( keepLooping === false )	return false;
100 	}
101 	return true;
102 };

104 /**
105  * Make a child Class inherit from the parent class.
106  *
107  * @param {Object} childClass the child class which gonna inherit
108  * @param {Object} parentClass the class which gonna be inherited
109 */
110 tQuery.inherit	= function(childClass, parentClass){
111 	// trick to avoid calling parentClass constructor
112 	var tempFn		= function() {};
113 	tempFn.prototype	= parentClass.prototype;
114 	childClass.prototype	= new tempFn();
115 
116 	childClass.prototype.parent	= parentClass.prototype;
117 	childClass.prototype.constructor= childClass;	
118 };

120 /**
121  * extend function. mainly aimed at handling default values - jme: im not sure at all it is the proper one.
122  * http://jsapi.info/_/extend
123  * similar to jquery one but much smaller
124 */
125 tQuery.extend = function(obj, base){
126 	var result	= {};
127 	base && Object.keys(base).forEach(function(key){
128 		result[key]	= base[key];
129 	})
130 	obj && Object.keys(obj).forEach(function(key){
131 		result[key]	= obj[key];
132 	})
133 	return result;
134 };

140 /**
141  * Make an object pluginable
142  * 
143  * @param {Object} object the object on which you mixin function
144  * @param {Object} dest the object in which to register the plugin
145 */
146 tQuery.pluginsOn	= function(object, dest){
147 	dest	= dest	|| object.prototype || object;
148 	object.register	= function(name, funct) {
149 		if( dest[name] ){
150 			throw new Error('Conflict! Already method called: ' + name);
151 		}
152 		dest[name]	= funct;
153 	};
154 	object.unregister	= function(name){
155 		if( dest.hasOwnProperty(name) === false ){
156 			throw new Error('Plugin not found: ' + name);
157 		}
158 		delete dest[name];
159 	};
160 	object.registered	= function(name){
161 		return dest.hasOwnProperty(name) === true;
162 	}
163 };
164 
165 tQuery.pluginsInstanceOn= function(klass){ return tQuery.pluginsOn(klass);		};
166 tQuery.pluginsStaticOn	= function(klass){ return tQuery.pluginsOn(klass. klass);	};
167 
168 
169 // make it pluginable
170 tQuery.pluginsOn(tQuery, tQuery);
