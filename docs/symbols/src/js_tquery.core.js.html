  5 /**
  6  * Create a tQuery element
  7  *
  8  * @class root class
  9  * 
 10  * - do something for crawling the three
 11  *   - like python.walk ?
 12  * - docs with jsdoc
 13  *   - http://www.thebrightlines.com/2010/05/06/new-template-for-jsdoctoolkit-codeview/
 14  *
 15  * 
 16  * @param {} object
 17  * @param {THREE.Object3D} rootnode
 18  * @returns {tQuery.*} the tQuery object created
 19 */
 20 var tQuery	= function(object, root)
 21 {
 22 	if( object instanceof THREE.Object3D  && tQuery.Object3D){
 23 		return new tQuery.Object3D(object);
 24 
 25 	}else if( object instanceof THREE.Geometry && tQuery.Geometry){
 26 		return new tQuery.Geometry(object);
 27 
 28 	}else if( object instanceof THREE.Material && tQuery.Material){
 29 		return new tQuery.Material(object);
 30 
 31 	}else if( typeof object === "string" && tQuery.Object3D){
 32 		return new tQuery.Object3D(object, root);
 33 
 34 	}else{
 35 		console.assert(false, "unsupported type")
 36 	}
 37 	return undefined;
 38 };

 49 /**
 50  * loop over a Array.
 51  * 
 52  * @param {Array} arr the array to traverse.
 53  * @param {Function} callback the function to notify. function(element){ }.
 54  * 			loop interrupted if it returns false
 55  * 
 56  * @returns {Boolean} return true if completed, false if interrupted
 57 */
 58 tQuery.each	= function(arr, callback){
 59 	for(var i = 0; i < arr.length; i++){
 60 		var keepLooping	= callback(arr[i])
 61 		if( keepLooping === false )	return false;
 62 	}
 63 	return true;
 64 };

 66 /**
 67  * Make a child Class inherit from the parent class.
 68  *
 69  * @param {Object} childClass the child class which gonna inherit
 70  * @param {Object} parentClass the class which gonna be inherited
 71 */
 72 tQuery.inherit	= function(childClass, parentClass){
 73 	var tempFn		= function() {};
 74 	tempFn.prototype	= parentClass.prototype;
 75 	childClass.prototype	= new tempFn();
 76 
 77 	childClass.prototype.parent	= parentClass.prototype;
 78 	childClass.prototype.constructor= childClass;	
 79 };

 85 /**
 86  * Make an object pluginable
 87  * 
 88  * @param {Object} object the object on which you mixin function
 89  * @param {Object} dest the object in which to register the plugin
 90 */
 91 tQuery.pluginsMixin	= function(object, dest){
 92 	dest	= dest	|| object.prototype || object;
 93 	object.register	= function(name, funct) {
 94 		if( dest[name] ){
 95 			throw new Error('Conflict! Already method called: ' + name);
 96 		}
 97 		dest[name]	= funct;
 98 	};
 99 	object.unregister	= function(name){
100 		if( dest.hasOwnProperty(name) === false ){
101 			throw new Error('Plugin not found: ' + name);
102 		}
103 		delete dest[name];
104 	};
105 	object.registered	= function(name){
106 		return dest.hasOwnProperty(name) === true;
107 	}
108 };
