 22 /**
 23  * contains the default material to use when create tQuery.Object3D
 24  * 
 25  * @fieldOf tQuery
 26  * @name defaultObject3DMaterial
 27 */
 28 tQuery.register('defaultObject3DMaterial', new THREE.MeshNormalMaterial());

 30 /**
 31  * Create a cube
 32  * 
 33  * @returns {tQuery.Object3D} a tQuery.Object3D containing it
 34 */
 35 tQuery.register('createCube', function(){
 36 	var ctor	= THREE.CubeGeometry;
 37 	var defaults	= [1, 1, 1, tQuery.defaultObject3DMaterial];
 38 	return this._createObj(ctor, defaults, arguments)
 39 });

 53 tQuery.register('_createObj', function(ctor, defaults, origArguments){
 54 	var args	= Array.prototype.slice.call( origArguments.length ? origArguments : defaults);
 55 
 56 	// init the material
 57 	var material	= tQuery.defaultObject3DMaterial;
 58 	// if the last arguments is a material, use it
 59 	if( args.length && args[args.length-1] instanceof THREE.Material ){
 60 		material	= args.pop();
 61 	}
 62 	
 63 	// ugly trick to get .apply() to work 
 64 	var createFn	= function(ctor, a0, a1, a2, a3, a4, a5, a6, a7){
 65 		console.assert(arguments.length <= 9);
 66 		return new ctor(a0,a1,a2,a3,a4,a5,a6,a7);
 67 	}
 68 	args.unshift(ctor);
 69 	var geometry	= createFn.apply(this, args);
 70 
 71 	// set the geometry.dynamic by default
 72 	geometry.dynamic= true;
 73 	// create the THREE.Mesh
 74 	var mesh	= new THREE.Mesh(geometry, material)
 75 	// return it
 76 	return tQuery(mesh);
 77 });

 79 tQuery.register('createAxis', function(){
