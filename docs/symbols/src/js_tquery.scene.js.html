  5 /**
  6  * Handle scene
  7  *
  8  * @class include THREE.Material
  9  *
 10  * @param {THREE.Material} object an instance or an array of instance
 11 */
 12 tQuery.Scene	= function()
 13 {
 14 	// update default scene.
 15 	// - TODO no sanity check ?
 16 	tQuery.scene	= this;
 17 	
 18 	// create a scene
 19 	this._scene	= new THREE.Scene();
 20 
 21 	// create a renderer
 22 	if( Detector.webgl ){
 23 		this._renderer = new THREE.WebGLRenderer({
 24 			antialias		: true,	// to get smoother output
 25 			preserveDrawingBuffer	: true	// to allow screenshot
 26 		});
 27 		this._renderer.setClearColorHex( 0xBBBBBB, 1 );
 28 	// uncomment if webgl is required
 29 	//}else{
 30 	//	Detector.addGetWebGLMessage();
 31 	//	return true;
 32 	}else{
 33 		this._renderer	= new THREE.CanvasRenderer();
 34 	}
 35 	// FIXME this window dimension is crap
 36 	this._renderer.setSize( window.innerWidth, window.innerHeight );
 37 
 38 	// create a camera in the scene
 39 	// FIXME this window dimension is crap
 40 	this._camera	= new THREE.PerspectiveCamera(35, window.innerWidth / window.innerHeight, 1, 10000 );
 41 	this._camera.position.set(0, 0, 5);
 42 	this._scene.add(this._camera);
 43 };
 44 
 45 // make it pluginable
 46 tQuery.pluginsMixin(tQuery.Scene);
 47 
 48 
 49 tQuery.Scene.prototype.destroy	= function()
 50 {
 51 	
 52 }

 58 tQuery.Scene.prototype.add	= function(object3d)
 59 {
 60 	if( object3d instanceof tQuery.Object3D ){
 61 		object3d.each(function(object3d){
 62 			this._scene.add(object3d)			
 63 		}.bind(this));
 64 	}else if( object3d instanceof THREE.Object3D ){
 65 		this._scene.add(object3d)		
 66 	}else	console.assert(false);
 67 	// for chained API
 68 	return this;
 69 }
 70 tQuery.Scene.prototype.remove	= function(object3d)
 71 {
 72 	console.assert(object3d instanceof THREE.Object3D)
 73 	this._scene.remove(object3d)
 74 	// for chained API
 75 	return this;
 76 }

 78 tQuery.Scene.prototype.fullpage	= function()
 79 {
 80 	// Should that be in pluging
 81 	var domElement	= document.body;
 82 	domElement.style.margin		= "0";
 83 	domElement.style.padding	= "0";
 84 	domElement.style.overflow	= 'hidden';
 85 	return this.appendTo(domElement);
 86 }
 87 
 88 tQuery.Scene.prototype.appendTo	= function(domElement)
 89 {
 90 	domElement.appendChild(this._renderer.domElement)
 91 	this._renderer.setSize( domElement.offsetWidth, domElement.offsetHeight );
 92 	// for chained API
 93 	return this;
 94 }
 95 
 96 tQuery.Scene.prototype.renderer	= function(){ return this._renderer;	}
 97 tQuery.Scene.prototype.camera	= function(){ return this._camera;	}
 98 tQuery.Scene.prototype.scene	= function(){ return this._scene;	}
 99 tQuery.Scene.prototype.get	= function(){ return this._scene;	}

105 tQuery.Scene.prototype.render	= function()
106 {
107 	// actually render the scene
108 	this._renderer.render( this._scene, this._camera );
